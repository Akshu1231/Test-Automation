var db = require('../../config/mysql');
 
 
// Open the DB connection, execute the sql query and close the db connection
function executeQuery(sql, callback) {
    db.connectionPool.getConnection((err,connection) => {
        if(err) {
            return callback(err, null);
        } else {
            console.log('DB Connected...')
            if(connection) {
                connection.query(sql, function (error, results) {
                    connection.release();
                    if (error) {
                        return callback(error, null);
                    } 
                    return callback(null, results);
                });
            }
        }
    });
}
 
// Select query builder function
function select(options={}, callback){
    let sql = "";
    let columns = "";
    let from = "";
    let joins = "";
    let conditions = "";
    let groupBy = "";
    let orderBy = "";
    let limit = "";
 
    let select = "SELECT";
 
    if(options.columns){columns = " " + options.columns;}else{columns = " *";}
    if(options.from){from = " FROM " + options.from;}
    if(options.joins){
        options.joins.forEach(join => {
            joins += " " + join.type + " JOIN " + join.table + " ON " + join.on;
        });
    }
    if(options.conditions){conditions = " WHERE 1 " + options.conditions;}else{conditions = " WHERE 1";}
    if(options.groupBy){groupBy = options.groupBy;}
    if(options.orderBy){
        let o = options.orderBy;
        orderBy += " ORDER BY " + o.by + " " + o.order;
    }
    if(options.limit){
        if(options.limit.limit && options.limit.limit > 0){
            let start = 0;
            if(options.limit.start){start = options.limit.start;}
 
            limit = " LIMIT " + start + "," + options.limit.limit;
        }        
    }
 
    if(!options.id){
        sql += select + " COUNT(*) as fcnt " + from + joins + conditions + groupBy + "; ";
        sql += select + " COUNT(*) as cnt " + from + joins + groupBy + "; ";
    }
 
    sql += select + columns + from + joins + conditions + groupBy + orderBy + limit;
 
    executeQuery(sql, function(err, data) {
        if(err) {
            return callback(err);
        }
        callback(null, data);
    });
}
 
// Insert query builder function
function insert(options={}, callback){
 
    let sql = "INSERT INTO " + options.table + " (" + options.columns + ") VALUES (" + options.values + ")";
 
    executeQuery(sql, function(err, data) {
        if(err) {
            return callback(err);
        }
        callback(null, data);
    });
}
 
// Update query builder function
function update(options={}, callback){
 
    let sql = "";
    if(options.conditions && options.conditions != ""){
        sql = "UPDATE " + options.table + " SET " + options.setValues + " WHERE " + options.conditions;
    }
 
    executeQuery(sql, function(err, data) {
        if(err) {
            return callback(err);
        }
        callback(null, data);
    });
}
 
// Remove query builder function (here we are updating is_removed field with 1)
function remove(options={}, callback){
 
    let sql = "";
    if(options.conditions && options.conditions != ""){
        sql = "UPDATE " + options.table + " SET is_removed = 1 WHERE " + options.conditions;
    }
 
    executeQuery(sql, function(err, data) {
        if(err) {
            return callback(err);
        }
        callback(null, data);
    });
}
 
// This function returns the values part of insert query
function insertString(val, delim){
    let str = "";
    if(typeof val == 'object'){
        str = delim + db.escape(JSON.stringify(val));
    }else{
        str = delim + db.escape(val);
    }
    return str;
}
 
// This function returns the set part of the update query
function updateString(key, val, delim){
    let str = "";
    if(typeof val == 'object'){
        str = delim + key + " = " + db.escape(JSON.stringify(val));
    }else{
        str = delim + key + " = " + db.escape(val);
    }
    return str;
}
 
// With this function we can directly execute the whole sql query in model, controller or services
function query(sql, callback) {    
    executeQuery(sql,function(err, data) {
        if(err) {
            return callback(err);
        }       
        callback(null, data);
    });
}
 
// This is used to append the single codes in sql query
function escape(data) {
    console.log("Type: ", data + typeof(data));
    if(typeof(data) == 'number' || typeof(data) === 'boolean'){
        return data;
    }else{
        return "'" + data + "'";
    }
}
 
// Here we are exporting all the above function to use in another files like model, controller and services
module.exports = {

    update: update,
    remove: remove,
    updateString: updateString,
    insertString: insertString,
    escape: escape

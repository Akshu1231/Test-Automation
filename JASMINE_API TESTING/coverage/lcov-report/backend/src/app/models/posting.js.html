var db = require('../utilities/sqlMapper');
var dbTable = "postings";
var tableAlias = " p ";
var dbFields = ['posting_id', 'enterprise_id', 'title', 'description, no_of_opening', 'required_exp', 'required_skills', 'posting_type', 'opening_date', 'expiry_date', 'location', 'address', 'qualification', 'duration', 'file', 'min_gpa', 'is_active', 'created_by', 'updated_by', 'is_remuneration', 'remuneration'];
 
 
function get(req, callback){  
    let conditions = "AND p.is_removed=0";
    let orderBy = {
        by: "p.updated_at",
        order: "DESC"
    };
    if(req.query.search){  
        conditions = ` AND (p.title LIKE '%` + req.query.search + `%' 
                            OR p.description   LIKE '%` + req.query.search + `%'
                            OR p.no_of_opening LIKE '%` + req.query.search + `%')`;
    }
 
    if(req.query.orderBy && req.query.orderBy != "" ){
        if(req.query.order && req.query.order != "" ){
            orderBy = {
                by : req.query.orderBy,
                order: req.query.order
            };
        }else {
            orderBy = {
                by : 'p.'+req.query.orderBy,
                order: "DESC"
            };
        }
     }
 
 
    let joins = [{
        type: "LEFT",
        table: "enterprise e ",
        on: "p.enterprise_id = e.enterprise_id"
    }];
 
    let options = {
        from: dbTable + tableAlias,
        conditions: conditions,
        joins:joins,
        orderBy:orderBy
    }
    db.select(options, function(err, result) {
        if(err) {
            return callback(err);
        }        
        callback(null, result);
    });
}
 
function getById(req, callback){    
    let options = {
        id: req.params.id,
        from: dbTable + tableAlias,
        conditions: " AND posting_id = " + req.params.id,
        columns: ""
    }
    db.select(options, function(err, result) {
        if(err) {
            return callback(err);
        }        
        callback(null, result);
    });
}
 
function add(req, callback){
    let data = req.body;
    let delim = "";
    let columns = "";
    let values = "";    
 
    for(let key in data){
		if(dbFields.includes(key)){
			columns += delim + key;
			values += db.insertString(data[key], delim);
			delim = ",";
		}
    }
    
    let options = {
        table: dbTable,
        columns: columns,
        values: values
    }
    db.insert(options, function(err, result) {
        if(err) {
            return callback(err);
        }        
        callback(null, result);
    });
}
 
function edit(req, callback){
    let data = req.body;
    let delim = "";
    let setValues = "";
    let conditions = "posting_id = " + req.params.id;
 
    for(let key in data){
		if(dbFields.includes(key)){
			setValues += db.updateString(key, data[key], delim);
			delim = ",";
		}
    }
    
    let options = {
        table: dbTable,
        setValues: setValues,
        conditions: conditions
    }
    db.update(options, function(err, result) {
        if(err) {
            return callback(err);
        }        
        callback(null, result);
    });
}
 
function remove(req, callback){
    let conditions = "posting_id = " + req.params.id;
    
    let options = {
        table: dbTable,
        conditions: conditions
    }
    db.remove(options, function(err, result) {
        if(err) {
            return callback(err);
        }        
        callback(null, result);
    });
}
 
module.exports = {
    get: get,
    getById: getById,
    add: add,
    edit: edit,
    remove: remove
}
